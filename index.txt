  <script>
    // Navegación principal
    const sections = [
      "home-section",
      "zones-section",
      "routines-section",
      "account-section",
      "settings-section",
      "nfc-section",
    ];

    const navButtons = document.querySelectorAll(".nav-btn");
    const sectionEls = sections.map((id) => document.getElementById(id));

    function showSection(id) {
      sectionEls.forEach((el) => {
        if (!el) return;
        el.classList.toggle("hidden", el.id !== id);
      });
    }

    navButtons.forEach((btn) => {
      btn.addEventListener("click", () => {
        const target = btn.getAttribute("data-target");
        if (!target) return;
        showSection(target);
        navButtons.forEach((b) => b.classList.remove("active"));
        btn.classList.add("active");
      });
    });

    // Abrir secciones desde elementos internos (por ejemplo, zonas, configuración, NFC)
    document.querySelectorAll("[data-open]").forEach((el) => {
      el.addEventListener("click", () => {
        const target = el.getAttribute("data-open");
        if (!target) return;
        showSection(target);
        // Desactivar todos los botones de nav si es una sección secundaria
        if (["settings-section", "nfc-section"].includes(target)) {
          navButtons.forEach((b) => b.classList.remove("active"));
        }
      });
    });

    // Toggles simples
    document.querySelectorAll(".toggle").forEach((toggle) => {
      toggle.addEventListener("click", () => {
        toggle.classList.toggle("on");
      });
    });

    // Simulación NFC
    const nfcButtons = document.querySelectorAll("[data-nfc]");
    const nfcToast = document.getElementById("nfc-toast");
    let nfcTimeout;

    function showNfcToast() {
      if (!nfcToast) return;
      nfcToast.classList.add("show");
      clearTimeout(nfcTimeout);
      nfcTimeout = setTimeout(() => {
        nfcToast.classList.remove("show");
      }, 2000);
    }

    nfcButtons.forEach((btn) => {
      btn.addEventListener("click", showNfcToast);
    });

    // Mostrar Home por defecto
    showSection("home-section");
  </script>
